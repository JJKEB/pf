GGIO2 틀린그림찾기 이벤트

담당파트 - 마크업, 스타일, 스크립트

( HTML, CSS, SASS, javascript, jQuery )

  • Responsive Web
- Logic -

1. 접속시 랜덤 이미지 변경
2. 틀린그림 부분을 클릭하면 O 다른부분을 클릭하면 X 표시
3. 모두 맞출시 "모두 찾으셨습니다." 글귀 출력

/* 틀린그림찾기
---------------------------------------------------------------- */
(function ($) {
  $(function () {
    // 틀린그림찾기 - 시작
    function otherF_imgFind() {
      // 참조변수
      var $otherF_imgWrap = $('.otherF_imgWrap'), // 비교 이미지 영역
        $otherNum = 3, // 틀린부분 갯수
        xBtn_Url = '/images/event_Picfind/ico-xMark.png', // x 마크 이미지 URL
        imgUrl = new Array(); // 틀린이미지 문제 URL
      imgUrl[0] = '/images/event_Picfind/img_otherF_0.jpg';
      imgUrl[1] = '/images/event_Picfind/img_otherF_1.jpg';
      imgUrl[2] = '/images/event_Picfind/img_otherF_2.jpg';

      // 랜덤 문제 변경
      var ranNum = Math.floor(Math.random() * imgUrl.length); // 랜덤 숫자
      $('.otherF_imgArea img').attr({
        src: imgUrl[ranNum],
      });

      // 문제 변경시 클래스 변경
      $otherF_imgWrap.addClass('otherF_Img' + ranNum);

      // 틀린부분 갯수 대로 생성 및 체크 표시 적용
      for (var i = 0; i < $otherNum; i++) {
        $otherF_imgWrap.append(
          ' <label for="ofCheck' +
            i +
            '" class="ofCheck' +
            i +
            '">ofCheck' +
            i +
            '</label><input type="radio" name="ofCheck' +
            i +
            '" id="ofCheck' +
            i +
            '" class="ofRadio"> ',
        );
        $otherF_imgWrap.append(
          '<div class="ofCheck' + i + '">ofCheck' + i + '</div>',
        );

        $('.ofCheck' + i).on('click', function () {
          var $bb = $(this).attr('class');
          var $cc = $('input#' + $bb + '');
          $cc.attr('checked', 'checked');
          $('.' + $bb).addClass('okFind');
        });
      }

      //  X 마크
      $otherF_imgWrap.append(
        '<div class="noAnswerX"><img src="' + xBtn_Url + '" alt="" /></div>',
      );
      var $noAnswerX = $('.noAnswerX'),
        $noAnswerW = $noAnswerX.width(),
        $noAnswerH = $noAnswerX.height();

      // 잘못 찾았을시  마우스 위치에 x 표시 및 모두 찾았을시 x 마크 삭제
      $('.otherF_imgArea').on('click', function (event) {
        event.stopPropagation();
        var $thisLeft = $(this).offset().left,
          $thisTop = $(this).offset().top,
          $x = event.pageX,
          $y = event.pageY;
        $noAnswerX
          .css({
            left: $x - ($thisLeft + $noAnswerW / 2),
            top: $y - ($thisTop + $noAnswerH / 2),
          })
          .css('display', 'block');
        $noAnswerX.delay(200).hide('scale', {}, 500);
      });

      // 모두 찾았을시
      $('input.ofRadio').on('click', function () {
        if ($('input.ofRadio:checked').length === $('input.ofRadio').length) {
          $noAnswerX.remove();
          alert('모두 찾으셨습니다.');
          return false;
        }
      });
    } // 틀린그림찾기 - 끝  ( otherF_imgFind )
    otherF_imgFind(); // 틀린그림찾기 함수 실행
  });
})(jQuery);
